int my_strncmp(const char* str1, const char* str2, size_t n)
{
    const char* end = str1+n;
    int result = 0;
    for(; result == 0 && str1 != end && (*str1 || *str2); result = *(str1++)-*(str2++));
    return result;
}

void search_substring(const char* substring, int file_count, ...) {
    va_list args;
    va_start(args, file_count);

    for (int i = 0; i < file_count; i++) {
        const char* file_path = va_arg(args, const char*);

        FILE* file = fopen(file_path, "r");

        if (file == NULL) {
//            printf("%s: %s\n", errorMessages[FILE_OPENING_ERROR], file_path);
            exit(FILE_OPENING_ERROR);
        }

        char line[100];
        int line_number = 1;

        while (fgets(line, sizeof(line), file) != NULL) {
            char* ptr = line;
            int index = 0;

            while (*ptr) {
                if (my_strncmp(ptr, substring, strlen(substring)) == 0) {
                    printf("Файл: %s | Строка: %d | Позиция: %d\n", file_path, line_number, index+1);
                }

                index++;
                ptr++;
            }

            line_number++;
        }

        if ( fclose (file) == EOF) {
//            printf("%s", errorMessages[FILE_CLOSING_ERROR]);
            exit(FILE_CLOSING_ERROR);
        }
    }

    va_end(args);
}
